This file shows how the different files and programming languages were compiled and runned for the binarytrees problem.

Java:
    -mv binarytrees.java-2.java binarytrees.java (because of class name)
    -javac -d . binarytrees.java
    -java binarytrees 12

JavaScript:
    -cp -L binarytrees.node binarytrees.js
    -node binarytrees.js 12

Python3:
    -python3 -OO binarytrees.py 12

PHP:
    -php -n binarytrees.php 12

C#:
    -mcs binarytrees.cs
    -mono binarytrees.exe 12

C:
  Compile:
    -(for 1)gcc -pipe -Wall -O3 -fomit-frame-pointer -march=native  binarytrees.gcc.c -o binarytrees.gcc_run -lm
    -(for 3)gcc -pipe -Wall -O3 -fomit-frame-pointer -march=native -fopenmp -D_FILE_OFFSET_BITS=64 -I/usr/include/apr-1.0 binarytrees.gcc-3.c -o binarytrees.gcc_run -lapr-1 -lgomp -lm
    -(for 5)gcc -pipe -Wall -O3 -fomit-frame-pointer -march=native -pthread binarytrees.gcc-5.c -o binarytrees.gcc_run
  Run:
    -./binarytrees.gcc_run 12

C++:
  Compile:
    -(for 1,3)g++ -c -pipe -O3 -fomit-frame-pointer -march=native   binarytrees.gpp.c++ -o binarytrees.c++.o && g++ binarytrees.c++.o -o binarytrees.gpp_run -lpthread
    -(for 2)g++ -c -pipe -O3 -fomit-frame-pointer -march=native   binarytrees.gpp-2.c++ -o binarytrees.c++.o && g++ binarytrees.c++.o -o binarytrees.gpp_run
    -(for 6,8)g++ -c -pipe -O3 -fomit-frame-pointer -march=native -fopenmp binarytrees.gpp-6.c++ -o binarytrees.c++.o && g++ binarytrees.c++.o -o binarytrees.gpp_run -fopenmp
    -(for 9)g++ -c -pipe -O3 -fomit-frame-pointer -march=native -fopenmp -I/usr/include/apr-1.0 binarytrees.gpp-9.c++ -o binarytrees.c++.o && g++ binarytrees.c++.o -o binarytrees.gpp_run -fopenmp -lapr-1
  Run:
    -./binarytrees.gpp_run 12

Ruby:
    -ruby -W0 binarytrees.yarv 12
